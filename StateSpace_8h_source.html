196 inline Scalar distance_to_rectangle(cref_t x, cref_t lb, cref_t ub) const {
197
198 assert(x(0) >= -M_PI);
199 assert(x(0) <= M_PI);
200
201 assert(lb(0) >= -M_PI);
202 assert(lb(0) <= M_PI);
203
204 assert(ub(0) >= -M_PI);
205 assert(ub(0) <= M_PI);
206
207 if (x(0) >= lb(0) && x(0) <= ub(0)) {
208 return 0;
209 } else if (x(0) > ub(0)) {
210 Scalar d1 = x(0) - ub(0);
211 Scalar d2 = lb(0) - (x(0) - 2 * M_PI);
212 assert(d2 >= 0);
213 assert(d1 >= 0);
214 return std::min(d1, d2);
215 } else if (x(0) < lb(0)) {
216 Scalar d1 = lb(0) - x(0);
217 Scalar d2 = (x(0) + 2 * M_PI) - ub(0);
218 assert(d2 >= 0);
219 assert(d1 >= 0);
220 return std::min(d1, d2);
221 } else {
222 assert(false);
223 return 0;
224 }
225 }

227 inline Scalar distance(cref_t x, cref_t y) const {
228
229 assert(x(0) >= -M_PI);
230 assert(y(0) >= -M_PI);
231
232 assert(x(0) <= M_PI);
233 assert(y(0) <= M_PI);
234
235 Scalar dif = x(0) - y(0);
236 if (dif > M_PI) {
237 dif -= 2 * M_PI;
238 } else if (dif < -M_PI) {
239 dif += 2 * M_PI;
240 }
241 Scalar out = std::abs(dif);
242 return out;
